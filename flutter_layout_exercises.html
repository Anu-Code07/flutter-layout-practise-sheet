Sticky day headers
SliverPersistentHeader · pinned: true
Itinerary grouped by Day 1 / Day 2 — section labels stay pinned at top while items scroll.

Create a SliverPersistentHeaderDelegate subclass. Add one per section group.

class _StickyHeader extends SliverPersistentHeaderDelegate {
  final String title;
  _StickyHeader(this.title);

  @override double get minExtent => 36;
  @override double get maxExtent => 36;

  @override
  Widget build(_, __, ___) => Container(
    color: Colors.white,
    alignment: Alignment.centerLeft,
    padding: EdgeInsets.symmetric(horizontal: 16),
    child: Text(title, style: TextStyle(fontWeight: FontWeight.w600)),
  );

  @override bool shouldRebuild(_) => false;
}

Anchored tooltip popup
Overlay · OverlayEntry · CompositedTransformFollower
Tap a price cell → floating tooltip with fare breakdown appears above it. Tap elsewhere to dismiss.

Use CompositedTransformTarget on the trigger and CompositedTransformFollower in the OverlayEntry.

final LayerLink _link = LayerLink();
OverlayEntry? _entry;

void _showTooltip() {
  _entry = OverlayEntry(builder: (_) =>
    Positioned.fill(
      child: GestureDetector(
        onTap: () { _entry?.remove(); _entry = null; },
        behavior: HitTestBehavior.translucent,
        child: CompositedTransformFollower(
          link: _link,
          offset: Offset(0, -80),
          child: Align(
            alignment: Alignment.topCenter,
            child: Material(elevation: 4, borderRadius: BorderRadius.circular(8), child: FareBreakdown()),
          ),
        ),
      ),
    ),
  );
  Overlay.of(context).insert(_entry!);
}

// Wrap trigger widget:
CompositedTransformTarget(link: _link, child: GestureDetector(onTap: _showTooltip, child: PriceCell()))

Custom radial layout
Flow · FlowDelegate
FlowDelegate
controls positions
FAB speed dial / radial menu — items animate from a central point. Each child's position computed per-frame.

Flow is repaint-efficient — only the delegate re-runs on animation tick, not full layout.

class _RadialDelegate extends FlowDelegate {
  final Animation animation;
  _RadialDelegate(this.animation) : super(repaint: animation);

  @override
  void paintChildren(FlowPaintingContext ctx) {
    for (int i = 0; i < ctx.childCount; i++) {
      final angle = i * (pi / 4); // 45° apart
      final x = animation.value * 80 * cos(angle);
      final y = animation.value * 80 * sin(angle);
      ctx.paintChild(i, transform: Matrix4.translationValues(x, y, 0));
    }
  }

  @override bool shouldRepaint(_) => true;
}

Flow(
  delegate: _RadialDelegate(animController),
  children: actions.map((a) => FABChild(a)).toList(),
)
